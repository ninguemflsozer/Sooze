CREATE TABLE alunos (
    id INT PRIMARY KEY,
    nome VARCHAR(100)
);

CREATE TABLE cursos (
    id INT PRIMARY KEY,
    nome VARCHAR(100)
);

CREATE TABLE matriculas (
    id INT PRIMARY KEY,
    aluno_id INT,
    curso_id INT
);

-- 1) Nome dos alunos com o nome do curso
SELECT 
    a.nome AS aluno,
    c.nome AS curso
FROM matriculas m
JOIN alunos a ON m.aluno_id = a.id
JOIN cursos c ON m.curso_id = c.id;

-- 2) Apenas alunos que possuem matrícula
SELECT 
    a.nome AS aluno
FROM matriculas m
JOIN alunos a ON m.aluno_id = a.id;

-- 3) Todos os alunos, mesmo sem matrícula
SELECT 
    a.nome AS aluno,
    c.nome AS curso
FROM alunos a
LEFT JOIN matriculas m ON a.id = m.aluno_id
LEFT JOIN cursos c ON m.curso_id = c.id;

-- 4) Alunos que NÃO estão matriculados
SELECT 
    a.nome AS aluno_sem_matricula
FROM alunos a
LEFT JOIN matriculas m ON a.id = m.aluno_id
WHERE m.aluno_id IS NULL;

-- 5) Nome do aluno e nome do curso
SELECT 
    a.nome AS aluno,
    c.nome AS curso
FROM matriculas m
JOIN alunos a ON m.aluno_id = a.id
JOIN cursos c ON m.curso_id = c.id;

-- 6) Apenas alunos do curso Informática
SELECT 
    a.nome AS aluno
FROM matriculas m
JOIN alunos a ON m.aluno_id = a.id
JOIN cursos c ON m.curso_id = c.id
WHERE c.nome = 'Informática';

-- 7) Quantidade de alunos por curso
SELECT 
    c.nome AS curso,
    COUNT(a.id) AS quantidade_alunos
FROM cursos c
LEFT JOIN matriculas m ON c.id = m.curso_id
LEFT JOIN alunos a ON m.aluno_id = a.id
GROUP BY c.nome
ORDER BY c.nome;
